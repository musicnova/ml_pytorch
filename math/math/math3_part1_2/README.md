import numpy as np

def marginal_a(P, a, b):
    # P -- matrix with probabilities, axis 0 -- xi, axis 1 -- eta
    # a -- array with outcomes for xi
    # b -- array with outcomes for eta
    # P.shape[0] == b.shape[0]
    # P.shape[1] == a.shape[0]
    p = 1.0 / (1.0 + np.exp(-P.dot(b) - a))
    da = a - p
    res = np.array(da).flatten()

    return res
